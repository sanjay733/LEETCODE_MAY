class Solution {
    public int[] longestObstacleCourseAtEachPosition(int[] obstacles) {
        int n=obstacles.length,n1=0;
        int []ans=new int[n];
        int []sub=new int[n];
        for(int i=0;i<n;i++){
            int j=find_index(sub,obstacles[i],n1);
            if(j==n1){
                n1++;
            }
            sub[j]=obstacles[i];
            ans[i]=j+1;
        }
        return ans;
    }
    private int find_index(int[] l, int t, int end){
        if(end==0){
            return 0;
        }
        int st=0;
        while(st<end){
            int mid=(st+end)/2;
            if(l[mid]<=t){
                st=mid+1;
            }
            else{
                end=mid;
            }
        }
        return st;
    } 
}
