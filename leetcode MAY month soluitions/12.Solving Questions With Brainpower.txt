class Solution {
    public long mostPoints(int[][] arr) {
        int n = arr.length;
        long[] dp = new long[n+1];
        Arrays.fill(dp,0);
        for(int i=n-1;i>=0;i--){
            long tem = 0;
            if(i+arr[i][1]+1 < n) tem = dp[i+arr[i][1]+1]; 
            long take = (long) arr[i][0] + tem;
            long nontake = dp[i+1];
            dp[i] = Math.max(take,nontake);
        }
        return dp[0];
    }
}
